# Build Common Includes

These snippets are shared across multiple commands to avoid duplication.

---

## Verify Project Structure

```bash
# Verify CMakeLists.txt exists
if [ ! -f "CMakeLists.txt" ]; then
    echo "❌ No CMakeLists.txt found. Are you in a JUCE plugin project?"
    exit 1
fi

# Confirm JUCE is available
if [ ! -d "JUCE" ] && [ ! -d "submodules/JUCE" ]; then
    echo "⚠️  JUCE framework not found. Set JUCE_PATH or add as submodule."
fi
```

## Configure CMake

```bash
echo "🔧 Configuring CMake..."

cmake -B build \
    -DCMAKE_BUILD_TYPE={{BUILD_TYPE}} \
    -DCMAKE_OSX_ARCHITECTURES="arm64;x86_64" \
    -DCMAKE_EXPORT_COMPILE_COMMANDS=ON

if [ $? -ne 0 ]; then
    echo "❌ CMake configuration failed"
    echo "Run cmake -B build for detailed error messages"
    exit 1
fi

echo "✓ CMake configuration successful"
```

## Detect Available Formats

```bash
echo ""
echo "📦 Detecting configured plugin formats..."

formats=()
if grep -q "VST3" build/CMakeCache.txt 2>/dev/null; then
    formats+=("VST3")
fi
if grep -q "AU" build/CMakeCache.txt 2>/dev/null; then
    formats+=("AU")
fi
if grep -q "AAX" build/CMakeCache.txt 2>/dev/null; then
    formats+=("AAX")
fi
if grep -q "Standalone" build/CMakeCache.txt 2>/dev/null; then
    formats+=("Standalone")
fi

if [ ${#formats[@]} -eq 0 ]; then
    echo "⚠️  No plugin formats detected"
else
    echo "Configured formats: ${formats[*]}"
fi
```

## Build All Targets

```bash
echo ""
echo "🏗️  Building all targets..."
echo "Build type: {{BUILD_TYPE}}"
echo "This may take several minutes..."

num_cores=$(sysctl -n hw.ncpu 2>/dev/null || nproc 2>/dev/null || echo 4)

cmake --build build \
    --config {{BUILD_TYPE}} \
    --parallel $num_cores

build_exit_code=$?

if [ $build_exit_code -ne 0 ]; then
    echo "❌ Build failed with exit code $build_exit_code"
    echo "Try: cmake --build build --verbose for details"
    exit 1
fi

echo "✓ Build completed successfully"
```

## Run Tests

```bash
echo ""
echo "🧪 Running tests..."

if [ -f "build/Tests/RunUnitTests" ] || [ -d "build/Tests" ]; then
    ctest --test-dir build -C {{BUILD_TYPE}} --output-on-failure
    test_exit_code=$?

    if [ $test_exit_code -eq 0 ]; then
        echo "✓ All tests passed"
    else
        echo "⚠️  Some tests failed"
    fi
else
    echo "ℹ️  No tests configured"
fi
```

## Locate Built Plugins

```bash
echo ""
echo "📍 Locating built plugins..."

plugin_name=$(grep "project(" CMakeLists.txt | head -1 | sed 's/.*(\(.*\)).*/\1/' | tr -d ' ')

# VST3
vst3_path=$(find build -name "*.vst3" -type d | head -1)
[ -n "$vst3_path" ] && echo "✓ VST3: $vst3_path ($(du -sh "$vst3_path" | cut -f1))"

# AU (macOS only)
au_path=$(find build -name "*.component" -type d | head -1)
[ -n "$au_path" ] && echo "✓ AU: $au_path ($(du -sh "$au_path" | cut -f1))"

# AAX
aax_path=$(find build -name "*.aaxplugin" -type d | head -1)
[ -n "$aax_path" ] && echo "✓ AAX: $aax_path ($(du -sh "$aax_path" | cut -f1))"

# Standalone
standalone_path=$(find build -name "${plugin_name}" -type f -o -name "${plugin_name}.app" -type d | head -1)
[ -n "$standalone_path" ] && echo "✓ Standalone: $standalone_path ($(du -sh "$standalone_path" | cut -f1))"
```

## Clean Build

```bash
echo "Clean previous build? (y/N)"
read -r response
if [[ "$response" =~ ^[Yy]$ ]]; then
    rm -rf build
    echo "✓ Cleaned build directory"
fi
```

## Determine Build Type

```bash
# Default to Release if not specified
BUILD_TYPE="${1:-Release}"

case "$BUILD_TYPE" in
    debug)
        BUILD_TYPE="Debug"
        ;;
    release|relwithdebinfo)
        BUILD_TYPE="Release"
        ;;
    *)
        echo "Unknown build type: $BUILD_TYPE"
        echo "Valid options: debug, release, relwithdebinfo"
        exit 1
        ;;
esac
```
